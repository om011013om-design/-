# Python Example import requests import base64 def convert_to_svg(image_path, api_key, options=None): url = "https://vectormagic-pro.api/v1/convert" headers = {"X-API-Key": api_key} with open(image_path, "rb") as f: image_data = base64.b64encode(f.read()).decode() payload = { "image": image_data, "colorScheme": "blackOnWhite", "threshold": 128, "strokeWidth": 2 } response = requests.post(url, headers=headers, json=payload) return response.json() # استخدم مفتاح API الخاص بك result = convert_to_svg("image.png", "YOUR_API_KEY") with open("output.svg", "w") as f: f.write(result["svg"])
// JavaScript / Node.js Example async function convertToSVG(imageFile, apiKey, options = {}) { const formData = new FormData(); formData.append('image', imageFile); formData.append('colorScheme', options.colorScheme || 'blackOnWhite'); formData.append('threshold', options.threshold || 128); const response = await fetch('https://vectormagic-pro.api/v1/convert', { method: 'POST', headers: { 'X-API-Key': apiKey }, body: formData }); return await response.json(); } // للاستخدام في بوت تليجرام bot.on('photo', async (ctx) => { const svg = await convertToSVG(photo, 'YOUR_API_KEY'); ctx.replyWithDocument({ source: Buffer.from(svg), filename: 'vector.svg' }); });
